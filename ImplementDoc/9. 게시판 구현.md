   ```react
   import React, { Component } from "react";
   import { Container, Nav, Button, Table, Row, Modal, Form, Badge } from "react-bootstrap";

       componentDidMount = async () => {
           this.LoadPosts("");
       }
   
       ////////////////////////////////////////////////////////////////////////////
       // Posts requrest
       LoadPosts = async (category) => {
           try {
   ```
